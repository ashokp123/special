const rejectBtn = document.getElementById("reject"); function moveButton() { const maxX = window.innerWidth - rejectBtn.offsetWidth - 20; const maxY = window.innerHeight - rejectBtn.offsetHeight - 20; const randomX = Math.floor(Math.random() * maxX); const randomY = Math.floor(Math.random() * maxY); rejectBtn.style.position = "fixed"; rejectBtn.style.left = randomX + "px"; rejectBtn.style.top = randomY + "px"; } // Move when mouse gets over it rejectBtn.addEventListener("mouseover", moveButton); // Move when user tries to click rejectBtn.addEventListener("click", function(e) { e.preventDefault(); moveButton(); }); // Move when finger touches it on mobile rejectBtn.addEventListener("touchstart", function(e) { e.preventDefault(); moveButton(); });
